### 2. Dirty File Protection
Before any AI changes, protect user work:

```bash
# Check for uncommitted changes
check_dirty() {
  if ! git diff --quiet || ! git diff --cached --quiet; then
    echo "⚠️  Dirty files detected"
    git stash push -m "claude-boris-auto-stash-$(date +%s)"
    echo "💾 User changes stashed"
  fi
}

# Restore after AI work
restore_user_changes() {
  if git stash list | grep -q "claude-boris-auto-stash"; then
    git stash pop
    echo "📂 User changes restored"
  fi
}
```

### 3. Undo Capability
Reverse the last AI-made change:

```bash
# Undo last commit (if made by Claude)
undo_last() {
  LAST_MSG=$(git log -1 --format="%s")
  if echo "$LAST_MSG" | grep -q "Claude Boris\|🤖"; then
    git reset --soft HEAD^
    echo "↩️  Reverted: $LAST_MSG"
  else
    echo "⚠️  Last commit not made by Claude Boris"
    echo "Use 'git reset --soft HEAD^' manually if intended"
  fi
}
```

### 4. Checkpoint System
Create named save points for easy rollback:

```bash
# Create checkpoint
create_checkpoint() {
  NAME=${1:-"checkpoint-$(date +%Y%m%d-%H%M%S)"}
  git stash push -m "checkpoint:$NAME"
  # Also create a tag for persistent checkpoints
  git tag -a "checkpoint/$NAME" -m "Checkpoint: $NAME"
  echo "📍 Checkpoint created: $NAME"
}

# List checkpoints
list_checkpoints() {
  echo "=== Stash Checkpoints ==="
  git stash list | grep "checkpoint:"
  echo ""
  echo "=== Tag Checkpoints ==="
  git tag -l "checkpoint/*"
}

# Restore checkpoint
restore_checkpoint() {
  NAME=$1
  if git stash list | grep -q "checkpoint:$NAME"; then
    git stash apply stash@{$(git stash list | grep -n "checkpoint:$NAME" | cut -d: -f1 | head -1)}
  elif git tag -l | grep -q "checkpoint/$NAME"; then
    git checkout "checkpoint/$NAME"
  else
    echo "❌ Checkpoint not found: $NAME"
  fi
}
```

## Safety Protocols

### Pre-Operation Checks
Before any destructive operation:

```bash
safety_check() {
  # 1. Check for uncommitted work
  if ! git diff --quiet HEAD; then
    echo "⚠️  Uncommitted changes detected"
    return 1
  fi

  # 2. Check we're not on protected branch
  BRANCH=$(git branch --show-current)
  if [[ "$BRANCH" == "main" || "$BRANCH" == "master" ]]; then
    echo "⚠️  On protected branch: $BRANCH"
    return 1
  fi

  # 3. Check for pending merge
  if [ -f .git/MERGE_HEAD ]; then
    echo "⚠️  Merge in progress"
    return 1
  fi

  return 0
}

# Verify push target
verify_push_target() {
  echo "Push targets:"
  git remote -v
  echo "Confirm the destination repo before pushing."
}
```

### Dangerous Command Protection
Block or warn on risky operations:

```bash
# These require explicit confirmation:
# - git push --force
# - git reset --hard
# - git clean -fd

# - git rebase on shared branches

dangerous_command() {

  echo "Type 'I understand' to proceed."
  # In practice, ask user for confirmation

```

### Branch Management
```bash
# Create feature branch from main
create_feature_branch() {
  NAME=$1
  git checkout main
  git pull origin main
  git checkout -b "feature/$NAME"
  echo "🌿 Created branch: feature/$NAME"
}

# Safe delete merged branches
cleanup_branches() {
  git branch --merged main | grep -v "main\|master" | xargs -r git branch -d
  echo "🧹 Deleted merged branches"
}
```

## Workflow Commands

### /undo
```bash
# Undo last Claude change
git log -1 --format="Last commit: %s by %an"

# If made by Claude
if git log -1 --format="%an" | grep -qi "claude"; then
  git reset --soft HEAD^
  echo "✅ Undone. Changes are now unstaged."
else
  echo "⚠️  Last commit was not by Claude Boris"
fi
```

### /checkpoint [name]
```bash
# Create named checkpoint
NAME=${1:-"auto-$(date +%H%M%S)"}
git add -A
git stash push -m "checkpoint:$NAME"
git tag "checkpoint/$NAME" 2>/dev/null || true
echo "📍 Checkpoint: $NAME"
echo "Restore with: /rollback $NAME"
```

### /rollback [name|steps]
```bash
# Rollback to checkpoint or N commits back
TARGET=$1

if [[ "$TARGET" =~ ^[0-9]+$ ]]; then
  # Numeric - rollback N commits
  git reset --soft HEAD~$TARGET
  echo "↩️  Rolled back $TARGET commits"
elif git tag -l | grep -q "checkpoint/$TARGET"; then
  # Named checkpoint tag
  git checkout "checkpoint/$TARGET"
  echo "↩️  Restored checkpoint: $TARGET"
else
  # Try stash
  STASH=$(git stash list | grep "checkpoint:$TARGET" | head -1 | cut -d: -f1)
  if [ -n "$STASH" ]; then
    git stash apply "$STASH"
    echo "↩️  Applied stash checkpoint: $TARGET"
  else
    echo "❌ Checkpoint not found: $TARGET"
  fi
fi
```
